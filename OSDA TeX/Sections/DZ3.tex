\begin{enumerate}

\item ac: $(ac)'' = abcde \neq ac \cup (a)'' \cup (c)'' = ac \cup ae \cup c = ace \rightarrow$ proper premise

\item $b: (b)'' = bc \rightarrow$ is pseudo-intent.
\end{enumerate}
